webfaction installs:

- Install python modules

modules:
django-bootstrap3
django-registration
south 

command:
$ PYTHONPATH=$HOME/webapps/care/lib/python3.3 easy_install-3.3 --install-dir=$HOME/webapps/care/lib/python3.3 --script-dir=$HOME/webapps/care/bin django-bootstrap3

- Create database for each app
userprofile
groupaccount
transaction
transactionreal
groupaacountinvite

command: 
$ python3.3 manage.py schemamigration <appname>

then create the database,
$ python3.3 manage.py syncdb

migrate all apps
$ python3.3 manage.py migrate <appname>


- Collect staticfiles
$ python3.3 manage.py collectstatic

read errors:
less ~/logs/user/error_care.log

apache restart:
~/webapps/malt/apache2/bin/restart
